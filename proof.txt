(* Proof of proposition I.I from Euclid's Elements. *)
Theorem prop1 :
(a : Point) -> (b : Point) -> (ab : Eq Point a b -> False) -> 
Exists Point (fun (c : Point) =>
    And (Eq Len (Length a b) (Length a c))
        (Eq Len (Length b c) (Length b a)))
