(* First Order Logic *)

(* Falsehood and Empty *)
Axiom Empty : Type
Axiom Empty.elim : (C: Type) -> Empty -> C
Axiom False : Prop
Axiom False.elim : (P: Prop) -> False -> P

(* Existential Quantifier *)
Axiom Exists : (A: Type) -> (A -> Prop) -> Prop
Axiom Exists.intro : (A: Type) -> (p: A -> Prop) -> (a: A) -> (h: p a) -> Exists A p
Axiom Exists.elim : (A: Type) -> (p: A -> Prop) -> (b: Prop) -> 
    (e: Exists A p) -> ((a: A) -> p a -> b) -> b

(* Universal Quanitifer *)
Axiom Forall : (A: Type) -> (A -> Prop) -> Prop
Axiom Forall.intro : (A: Type) -> (p: A -> Prop) ->
    ((a: A) -> p a) -> Forall A p
Axiom Forall.elim : (A: Type) -> (p: A -> Prop) ->
    Forall A p -> (a: A) -> p a

(* Conjunction *)
Axiom And : Prop -> Prop -> Prop
Axiom And.intro : (A : Prop) -> (B : Prop) -> (a : A) -> (b : B) -> And A B
Axiom And.elim : (A : Prop) -> (B : Prop) -> (C : Prop) ->
    (f : A -> B -> C) -> And A B -> C

(* Disjunction *)
Axiom Or : Prop -> Prop -> Prop
Axiom Or.inl : (A : Prop) -> (B : Prop) -> A -> Or A B
Axiom Or.inr : (A : Prop) -> (B : Prop) -> B -> Or A B
Axiom Or.elim : (A : Prop) -> (B : Prop) -> (C : Prop) ->
    Or A B -> (A -> C) -> (B -> C) -> C

(* Equality *)
Axiom Eq : (T: Type) -> T -> T -> Prop
Axiom Eq.intro : (T: Type) -> (t: T) -> Eq T t t
Axiom Eq.elim : (T: Type) -> (t: T) -> (motive : T -> Prop) -> (rfl: motive t) -> (t1: T) -> Eq T t t1 -> motive t1

(* System for Geometry *)

(* Primitives *)
Axiom Point : Type
Axiom Line : Type
Axiom Circle : Type

(* Length (Len) Axioms (for any magnitude) *)
Axiom Len : Type
(* There exists a total order "Lt" on Len *)
Axiom Lt : Len -> Len -> Prop
(* Transitivity *)
Axiom LtTrans : (a: Len) -> (b: Len) -> (c: Len) -> Lt a b -> Lt b c -> Lt a c
(* Trichotomy *)
Axiom LtTricot : (a: Len) -> (b: Len) -> Or (Lt a b) (Or (Lt b a) (Eq _ a b))
Axiom LtAntisymm : (a: Len) -> (b: Len) -> Lt a b -> Lt b a -> False
Theorem LtNe : (a: Len) -> (b: Len) -> Lt a b -> Eq _ a b -> False :=
  fun (a: Len) (b: Len) (lt_ab: Lt a b) (eq_ab: Eq _ a b) =>
    LtAntisymm b b
      (Eq.elim Len a (fun (x: Len) => Lt x b) lt_ab b eq_ab)
      (Eq.elim Len a (fun (x: Len) => Lt x b) lt_ab b eq_ab)
(* There exists an element zero of len that is the least of sort len *)
Axiom Zero : Len
Axiom ZeroLeast : (a: Len) -> Or (Lt Zero a) (Eq _ Zero a)
(* There is an operation Add on Len which is commutative and associative and has zero as identity *)
Axiom Add : Len -> Len -> Len
Axiom AddComm : (a: Len) -> (b: Len) -> Eq _ (Add a b) (Add b a)
Axiom AddAssoc : (a: Len) -> (b: Len) -> (c: Len) -> Eq _ (Add (Add a b) c) (Add a (Add b c))
Axiom AddZero : (a: Len) -> Eq _ (Add a Zero) a
(* Add respects the order Lt *)
Axiom LtAdd : (a: Len) -> (b: Len) -> (c: Len) -> Lt a b -> Lt (Add a c) (Add b c)

(* Basic Relations *)

(* A point being on a line *)
Axiom OnLine : Point -> Line -> Prop

(* Two points on the same side as a line *)
Axiom SameSide : Point -> Point -> Line -> Prop

(* A point is between two points, all of which are distinct and collinear *)
Axiom Between : Point -> Point -> Point -> Prop

(*A point being on a circle*)
Axiom OnCircle : Point -> Circle -> Prop

(*A point being inside the circle*)
Axiom InCircle : Point -> Circle -> Prop

(*A point being the center of a circle*)
Axiom CenterCircle : Point -> Circle -> Prop

(* Two lines intersect *)
Axiom LinesInter : Line -> Line -> Prop

(* A line and a circle intersect *)
Axiom LineCircleInter : Line -> Circle -> Prop

(*Two circles intersect*)
Axiom CirclesInter : Circle -> Circle -> Prop

(* There is a length between all pairs of points *)
Axiom Length : (a: Point) -> (b: Point) -> Len

(* There is an angle for every three points *)
Axiom Angle : (a: Point) -> (b: Point) -> (c: Point) -> Len

(* There is an area of the triangle formed by three points *)
Axiom Area : (a: Point) -> (b: Point) -> (c: Point) -> Len

(* Constant representing right angle *)
Axiom RightAngle : Len

(* Let a be a point on a line L *)
Axiom pt_on_line :
  (L : Line) ->
  Exists Point (fun (a : Point) => OnLine a L)

(* Let a be a point on line L between b and c *)
Axiom pt_between_on_line :
  (L : Line) -> (b : Point) -> (c : Point) ->
  OnLine b L -> OnLine c L -> (Eq Point b c -> False) ->
  Exists Point (fun (a : Point) =>
    And (OnLine a L) (Between b a c))

(* Let a be a point on line L extending segment bc *)
Axiom pt_extension_on_line :
  (L : Line) -> (b : Point) -> (c : Point) ->
  OnLine b L -> OnLine c L -> (Eq Point b c -> False) ->
  Exists Point (fun (a : Point) =>
    And (OnLine a L) (Between b c a))

(* Let a be a point on the same side of L as b *)
Axiom pt_sameside_of_not_online :
  (L : Line) -> (b : Point) ->
  (OnLine b L -> False) ->
  Exists Point (fun (a : Point) => SameSide a b L)

(* Let a be a point on the opposite side of L from b *)
Axiom pt_oppositeside_of_not_online :
  (L : Line) -> (b : Point) ->
  (OnLine b L -> False) ->
  Exists Point (fun (a : Point) =>
    And (OnLine a L -> False)
        (SameSide a b L -> False))

(* Let a be a point on circle aa *)
Axiom pt_on_circle :
  (aa : Circle) ->
  Exists Point (fun (a : Point) => OnCircle a aa)

(* Let a be a point inside circle aa *)
Axiom pt_inside_circle :
  (aa : Circle) ->
  Exists Point (fun (a : Point) => InCircle a aa)

(* Let a be a point outside circle aa *)
Axiom pt_outside_circle :
  (aa : Circle) ->
  Exists Point (fun (a : Point) =>
    And (OnCircle a aa -> False)
        (InCircle a aa -> False))

(* Lines and Circles Construction Rules *)

(* Let L be the line through distinct points a and b *)
Axiom line_of_pts :
  (a : Point) -> (b : Point) -> (Eq Point a b -> False) ->
  Exists Line (fun (L : Line) =>
    And (OnLine a L) (OnLine b L))

(* Let aa be the circle centered at a and passing through b *)
Axiom circle_of_ne :
  (a : Point) -> (b : Point) -> (Eq Point a b -> False) ->
  Exists Circle (fun (aa : Circle) =>
    And (CenterCircle a aa) (OnCircle b aa))

(* Intersection Construction Rules *)

(* Let a be a point of intersection of lines L and M *)
Axiom pt_of_linesinter :
  (L : Line) -> (M : Line) -> LinesInter L M ->
  Exists Point (fun (a : Point) =>
    And (OnLine a L) (OnLine a M))

(* Let a be a point of intersection of line L and circle aa *)
Axiom pt_of_linecircleinter :
  (L : Line) -> (aa : Circle) -> LineCircleInter L aa ->
  Exists Point (fun (a : Point) =>
    And (OnLine a L) (OnCircle a aa))

(* Let a,b be the two points of intersection of line L and circle aa *)
Axiom pts_of_linecircleinter :
  (L : Line) -> (aa : Circle) -> LineCircleInter L aa ->
  Exists Point (fun (a : Point) =>
    Exists Point (fun (b : Point) =>
      And (Eq Point a b -> False)
      (And (OnLine a L)
      (And (OnLine b L)
      (And (OnCircle a aa) (OnCircle b aa))))))

(* Let a be a point on circle aa on line L between b and c *)
Axiom pt_oncircle_between_inside_outside :
  (L : Line) -> (aa : Circle) -> (b : Point) -> (c : Point) ->
  OnLine b L -> OnLine c L ->
  InCircle b aa ->
  (OnCircle c aa -> False) -> (InCircle c aa -> False) ->
  Exists Point (fun (a : Point) =>
    And (OnLine a L)
    (And (OnCircle a aa) (Between b a c)))

(* Let a be a point of intersection of circle aa and line L such that a,b,c are in order a-b-c,
   given b is inside aa and c neq b with both on L *)
Axiom pt_oncircle_extension_from_inside :
  (L : Line) -> (aa : Circle) -> (b : Point) -> (c : Point) ->
  OnLine b L -> OnLine c L ->
  InCircle b aa ->
  (Eq Point c b -> False) ->
  Exists Point (fun (a : Point) =>
    And (OnLine a L)
    (And (OnCircle a aa) (Between a b c)))

(* Let a be a point of intersection of circles aa and bb *)
Axiom pt_of_circlesinter :
  (aa : Circle) -> (bb : Circle) -> CirclesInter aa bb ->
  Exists Point (fun (a : Point) =>
    And (OnCircle a aa) (OnCircle a bb))

(* Let a,b be the two points of intersection of circles aa and bb *)
Axiom pts_of_circlesinter :
  (aa : Circle) -> (bb : Circle) -> CirclesInter aa bb ->
  Exists Point (fun (a : Point) =>
    Exists Point (fun (b : Point) =>
      And (Eq Point a b -> False)
      (And (OnCircle a aa)
      (And (OnCircle a bb)
      (And (OnCircle b aa) (OnCircle b bb))))))

(* Let a be a point of intersection of circles aa and bb on the same side of L as b,
   where L passes through centers c,d of aa and bb respectively *)
Axiom pt_sameside_of_circlesinter :
  (b : Point) -> (c : Point) -> (d : Point) -> (L : Line) -> (aa : Circle) -> (bb : Circle) ->
  OnLine c L -> OnLine d L -> (OnLine b L -> False) ->
  CenterCircle c aa -> CenterCircle d bb -> CirclesInter aa bb ->
  Exists Point (fun (a : Point) =>
    And (SameSide a b L)
    (And (OnCircle a aa) (OnCircle a bb)))

(* Let a be a point of intersection of circles aa and bb on the opposite side of L from b,
   under the same center/line hypotheses. *)
Axiom pt_oppositeside_of_circlesinter :
  (b : Point) -> (c : Point) -> (d : Point) -> (L : Line) -> (aa : Circle) -> (bb : Circle) ->
  OnLine c L -> OnLine d L -> (OnLine b L -> False) ->
  CenterCircle c aa -> CenterCircle d bb -> CirclesInter aa bb ->
  Exists Point (fun (a : Point) =>
    And (OnLine a L -> False)
    (And (SameSide a b L -> False)
         (And (OnCircle a aa) (OnCircle a bb))))

(* A length is zero iff the points are equal *)
Axiom length_eq_zero_iff : (a: Point) -> (b: Point) -> 
  And (Eq _ (Length a b) Zero -> Eq _ a b)
      (Eq _ a b -> Eq _ (Length a b) Zero)

(* Between is symmetric *)
Axiom between_symm : (a: Point) -> (b: Point) -> (c: Point) -> Between a b c -> Between c b a

Axiom length_symm : (a: Point) -> (b: Point) -> Eq _ (Length a b) (Length b a)

Axiom ne_13_of_between : (a: Point) -> (b: Point) -> (c: Point) -> Between a b c -> (Eq _ a c -> False)
Axiom ne_23_of_between : (a: Point) -> (b: Point) -> (c: Point) -> Between a b c -> (Eq _ b c -> False)

Axiom length_sum_of_between : (a: Point) -> (b: Point) -> (c: Point) -> Between a b c -> Eq _ (Add (Length a b) (Length b c)) (Length a c)

Axiom in_circle_iff_length_lt : (a: Point) -> (b: Point) -> (c: Point) -> (aa: Circle) -> 
  CenterCircle a aa -> OnCircle b aa -> 
  And (Lt (Length a c) (Length a b) -> InCircle c aa)
      (InCircle c aa -> Lt (Length a c) (Length a b))
